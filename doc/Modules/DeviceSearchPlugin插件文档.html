插件简介：
//插件名称：DeviceSearchPlugin插件
//插件功能：搜索DVR,IPC设备，修改IPC的网络参数

接口说明：
//简介：开始搜索设备
//返回值：
//      0:调用成功
int Start();

接口说明：
//简介：停止搜索设备
//返回值：
//      0:调用成功
int Stop();

接口说明：
//简介：刷新列表
//返回值：
//      0:调用成功
int Flush();

接口说明：
//简介：设置搜索时间间隔
//输入参数：
//      nInterval:时间间隔
//返回值：
//      0:调用成功
int setInterval(int nInterval);

接口说明：
//简介：设置网络参数
//输入参数：
//      sDeviceID:设备ID
//      sAddress:设备IP
//      sMask:子网掩码
//      sGateway:网关
//      sMac:MAC地址
//      sPort:端口号
//      sUsername:用户名
//      sPassword:密码
//返回值：
//      0:调用成功
//      1:调用失败
int SetNetworkInfo(const QString &sDeviceID,
const QString &sAddress,
const QString &sMask,
const QString &sGateway,
const QString &sMac,
const QString &sPort,
const QString &sUsername,
const QString &sPassword);

接口说明：
//简介：一键设置ip
//输入参数：
// 无   ；

//AutoSetNetworkInfoID(一键设置ip的设备信息id)，格式如下：  

<devnetworkInfo Num="2">
<dev sDeviceID="" sAddress="" sMask="" sGateway="" sMac="" sPort="" sUsername="" sPassword=""/>
<dev sDeviceID="" sAddress="" sMask="" sGateway="" sMac="" sPort="" sUsername="" sPassword=""/>
</devnetworkInfo>
//返回值：
//      0:调用成功
int AutoSetNetworkInfo();

插件抛出的事件：
@1
事件名称：SettingStatus
事件参数:"Status":
//Status可选值:
//      “set info success”:设置参数成功
//      “set port success”:设置端口成功
//      “Unauthorized”:用户名不正确
