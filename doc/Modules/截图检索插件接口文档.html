插件简介：
//插件名称：ScreenShotSearchPlugin插件
//插件功能：检索截图信息

接口说明：

//简述：根据输入参数检索截图信息。
//输入参数:
//sWndId: 窗口号(如111表示仅检索0，1，2窗口)
// sStartTime: 开始时间(格式为"YYYY-MM-DD")
// sEndTime: 结束时间(格式为"YYYY-MM-DD")
// nType:类型(按位表示，001：预览，010：本地回放，100：远程回放， 111表示所有类型)
// sUser:用户名，当前用户名
//返回值:
// 0: 调用成功
// 1: 参数错误
// 2: 系统错误
int searchScreenShot(const QString &sWndId,const QString &sStartTime,const QString &sEndTime,const int &nType,const QString &sUser);

//简述：获取截图文件总数
//返回值为截图文件数量
int getImageNum();

//简述：获取指定文件信息
//输入参数：
//nStartIndex:开始序号
//nEndIndex：结束序号
//返回值为截图文件信息
//格式：
{"fileDir":"D:/screenShotEx","fileName":"admin-3-0-2015-02-13-14-41-50-475.jpg","time":"2015-02-13 14:41:50","type":"0","userName":"admin","wndId":"3"},
{"fileDir":"D:/screenShotEx","fileName":"admin-4-0-2015-02-13-14-41-55-369.jpg","time":"2015-02-13 14:41:55","type":"0","userName":"admin","wndId":"4"},
{"fileDir":"D:/screenShotEx","fileName":"admin-1-0-2015-02-13-14-41-57-844.jpg","time":"2015-02-13 14:41:57","type":"0","userName":"admin","wndId":"1"}
"fileName"：文件名称
"fileDir"：文件路径
"wndId":窗口号(1-49)
"type"：类型(0：预览，1：本地回放，2：远程回放)
"time":截图时间(格式为"YYYY-MM-DD hh:mm:ss")
"userName"：用户名
QString getImageInfo(const int &nStartIndex, const int &nEndIndex);
