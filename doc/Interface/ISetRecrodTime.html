接口简介：
//名称：录像时间设置接口
//功能：提供录像时间设置接口
表名称：recordtime
字段解析：
id：integer primary key autoincrement
chl_id:integer(通道id)
schedle_id:integer(时间段id)
weekday:integer（day）
starttime：char（64）（开始录像时间）
endtime：char（64）（结束录像时间）
enable:integer（按位计算，00：没有录像，01：定时录像，10：移动录像，11：定时与移动录像）

create table recordtime
(
id integer primary key autoincrement,
chl_id integer,
schedle_id integer,
weekday integer,
starttime char（64）,
endtime char（64）,
enable integer
);

包含同文件ISetRecordTime.h

接口定义：

//修改设备录像时间
//输入参数：
// recordtime_id：指定需修改的录像时间的id号
// starttime：开始录像时间(时间格式："yyyy-MM-dd hh:mm:ss")
// endtime：结束录像时间(时间格式："yyyy-MM-dd hh:mm:ss")
// enable：按位计算，00：没有录像，01：定时录像，10：移动录像，11：定时与移动录像
//返回值：
// 0：修改成功
// 1：修改失败
int ModifyRecordTime(int recordtime_id,QString starttime,QString endtime,int enable）

//查询指定通道下的录像时间记录
//输入参数：
// chl_id：通道号
//返回值：
// 录像时间记录的id号列表
QStringList GetRecordTimeBydevId(int chl_id)

//查询指定录像时间记录条的字段值
//输入参数：
// recordtime_id：录像时间记录id
//返回值：
// 录像时间的字段值：
// chl_id:通道号
// schedle_id：时间段id（可选值0，1，2，3）
// weekday：（可选值 0（周一），1（周二），2（周三），3（周四），4（周五），5（周六），6（星期天））
// starttime：开始时间(时间格式："yyyy-MM-dd hh:mm:ss")
// endtime：结束时间(时间格式："yyyy-MM-dd hh:mm:ss")
// enable：按位计算，00：没有录像，01：定时录像，10：移动录像，11：定时与移动录像
QVariantMap GetRecordTimeInfo(int recordtime_id)
